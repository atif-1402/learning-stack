## 1. Polling Loop (external command)

Run a command every N seconds and display output.

```qml
Text {
  id: display
  property string data: "loading..."

  Process {
    id: poller
    command: ["your-command", "--arg"]
    running: true
    stdout: StdioCollector {
      onStreamFinished: display.data = this.text.trim()
    }
  }

  Timer {
    interval: 5000  // 5 seconds
    running: true
    repeat: true
    onTriggered: poller.running = true
  }

  text: data
}
```

**Used in:** [[17. Bar Component]] — BatteryWidget, NetworkWidget, BluetoothWidget, etc.

## 2. Multi-Monitor Setup

Create one window per connected screen.

```qml
ShellRoot {
  Variants {
    model: Quickshell.screens
    PanelWindow {
      required property var modelData
      screen: modelData
      anchors { top: true; left: true; right: true }
      implicitHeight: 28
      // ... bar content
    }
  }
}
```

**Used in:** All `shell.qml` files

## 3. Shared Timer + Process (efficient)

Keep one Timer and Process per config, share via property.

```qml
Scope {
  id: root
  property string timeData

  Process {
    id: timeProc
    command: ["date"]
    running: true
    stdout: StdioCollector {
      onStreamFinished: root.timeData = this.text
    }
  }

  Timer {
    interval: 1000
    running: true
    repeat: true
    onTriggered: timeProc.running = true
  }

  Variants {
    model: Quickshell.screens
    PanelWindow {
      Text { text: root.timeData }   // all screens show same value
    }
  }
}
```

**Used in:** [[17. Bar Component]] (shared theme polling)

## 4. Singleton Theme Provider

Global colors accessible from any widget.

```qml
// Theme.qml
pragma Singleton
import Quickshell
Singleton {
  property color background: "#1e1e2e"
  property color foreground: "#cdd6f4"
  function fg(a) { return Qt.rgba(foreground.r, foreground.g, foreground.b, a) }
}

// AnyWidget.qml
Rectangle { color: Theme.background }
Text { color: Theme.fg(0.8) }
```

**Used in:** [[22. Theme System and Omarchy Integration]]

## 5. Animated Popup (Tooltip)

Show/hide a tooltip with bounce animation.

```qml
PopupWindow {
  id: tooltip
  visible: false

  PopupAnchor {
    anchors {
      window: parent
      edge: Qt.BottomEdge
    }
  }

  Behavior on opacity {
    NumberAnimation { duration: 150 }
  }

  Behavior on y {
    NumberAnimation { easing: Easing.OutBack; duration: 300 }
  }

  function show() {
    tooltip.visible = true
    tooltip.opacity = 1
  }

  function hide() {
    tooltip.opacity = 0
    // hide after fade completes
  }
}
```

**Used in:** [[17. Bar Component]] — BarTooltip

## 6. Clickable Icon with Commands

Nerd Font icon with left/right click actions.

```qml
Item {
  property string icon
  property string leftCommand
  property string rightCommand

  Text {
    text: parent.icon
    font.family: "JetBrainsMono Nerd Font Propo"
    color: Theme.foreground
  }

  MouseArea {
    anchors.fill: parent
    hoverEnabled: true
    acceptedButtons: Qt.LeftButton | Qt.RightButton

    onClicked: (mouse) => {
      if (mouse.button === Qt.RightButton) {
        CommandRunner.run(rightCommand)
      } else {
        CommandRunner.run(leftCommand)
      }
    }
  }
}
```

**Used in:** [[17. Bar Component]] — BarIcon

## 7. Named Pipe IPC (OSD)

Send messages from shell scripts to QML.

```qml
// Writer (bash script)
#!/bin/bash
echo "volume:45" > /tmp/qs-osd-pipe

// Reader (QML)
Socket {
  id: pipe
  path: "/tmp/qs-osd-pipe"
  onData: {
    const parts = data.split(":")
    const type = parts[0]
    const value = parts[1]
    // update UI based on type + value
  }
}
```

**Used in:** [[19. OSD Component]], [[23. IPC and Named Pipes]]

## 8. Canvas Drawing (custom shapes)

Draw anything with 2D context.

```qml
Canvas {
  width: 100; height: 100
  onPaint: {
    const ctx = getContext("2d")
    ctx.fillStyle = "#1e1e2e"
    ctx.fillRect(0, 0, width, height)

    ctx.globalCompositeOperation = "destination-out"
    ctx.fillStyle = "#000000"
    ctx.beginPath()
    ctx.roundRect(10, 10, 80, 80, 20, 20)
    ctx.fill()
  }
}
```

**Used in:** [[20. Frame Component]]

## 10. File Watching

React to config file changes.

```qml
FileView {
  path: "config.json"
  onModified: {
    // File changed — reload
    const data = JsonAdapter.parse(content)
    root.radius = data.radius
  }
}
```
