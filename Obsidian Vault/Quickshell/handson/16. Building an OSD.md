```qml
import Quickshell
import Quickshell.Io
import Quickshell.Wayland
import QtQuick

PanelWindow {
  required property var modelData
  screen: modelData

  anchors { bottom: parent.bottom; horizontalCenter: parent.horizontalCenter }
  implicitWidth: 300
  implicitHeight: 0   // starts hidden
  color: "transparent"

  WlrLayershell.layer: WlrLayer.Overlay
  exclusiveZone: -1  // don't reserve space

  // --- State ---
  property string osdType: ""
  property int osdValue: 0
  property string osdText: ""
  property bool osdMuted: false
  property bool osdVisible: false

  // --- Background pill ---
  Rectangle {
    id: pill
    anchors.centerIn: parent
    width: parent.implicitWidth
    height: osdVisible ? 50 : 0
    radius: 25
    color: Qt.rgba(0.11, 0.11, 0.15, 0.9)

    Behavior on height {
      NumberAnimation { duration: 150; easing: Easing.OutCubic }
    }
    Behavior on opacity {
      NumberAnimation { duration: 150 }
    }

    // Icon
    Text {
      id: iconText
      anchors { left: parent.left; leftMargin: 16; verticalCenter: parent.verticalCenter }
      font.family: "JetBrainsMono Nerd Font Propo"
      font.pixelSize: 24
      color: osdMuted ? "#f38ba8" : "#cdd6f4"
    }

    // Progress bar
    Rectangle {
      id: barBg
      anchors { left: iconText.right; leftMargin: 12; verticalCenter: parent.verticalCenter }
      width: 150; height: 6
      radius: 3
      color: "#45475a"

      Rectangle {
        id: barFill
        width: parent.width * (osdValue / 100)
        height: parent.height
        radius: 3
        color: osdMuted ? "#f38ba8" : "#fab387"

        Behavior on width {
          NumberAnimation { duration: 100 }
        }
      }
    }

    // Label
    Text {
      id: labelText
      anchors { left: barBg.right; leftMargin: 12; verticalCenter: parent.verticalCenter }
      color: osdMuted ? "#f38ba8" : "#a6adc8"
      font.pixelSize: 14
    }
  }

  // --- Named pipe reader ---
  Process {
    id: pipeReader
    command: ["bash", "-c", "cat /tmp/qs-osd-pipe 2>/dev/null || sleep 1"]
    running: true
    stdout: StdioCollector {
      onStreamFinished: {
        const msg = this.text.trim()
        if (msg === "") return
        parseMessage(msg)
        // Re-run immediately for next message
        pipeReader.running = true
      }
    }
  }

  // --- Message parser ---
  function parseMessage(msg: string) {
    const parts = msg.split(":")
    const type = parts[0]
    const value = parts[1]

    switch (type) {
      case "volume":
        osdType = "volume"
        iconText.text = "\uf028"
        if (value === "muted") {
          osdMuted = true
          osdValue = 0
          labelText.text = "Muted"
        } else {
          osdMuted = false
          osdValue = parseInt(value)
          labelText.text = value + "%"
        }
        break
      case "brightness":
        osdType = "brightness"
        iconText.text = "\uf185"
        osdMuted = false
        osdValue = parseInt(value)
        labelText.text = value + "%"
        break
      case "mic":
        osdType = "mic"
        iconText.text = "\uf130"
        if (value === "muted") {
          osdMuted = true
          labelText.text = "Muted"
        } else {
          osdMuted = false
          labelText.text = "Active"
        }
        break
    }

    showOSD()
  }

  // --- Show/hide logic ---
  Timer {
    id: hideTimer
    interval: 2000
    onTriggered: osdVisible = false
  }

  function showOSD() {
    osdVisible = true
    hideTimer.restart()
  }
}
```

## Step 3: Create trigger scripts

```bash
nano ~/qs-tutorial/myosd/scripts/osd-volume
```

```bash
#!/bin/bash
pipe=/tmp/qs-osd-pipe

# Create pipe if it doesn't exist
[ -p "$pipe" ] || mkfifo "$pipe"

case "$1" in
  up)
    wpctl set-volume @DEFAULT_AUDIO_SINK@ 5%+
    ;;
  down)
    wpctl set-volume @DEFAULT_AUDIO_SINK@ 5%-
    ;;
  mute)
    wpctl set-mute @DEFAULT_AUDIO_SINK@ toggle
    ;;
esac

# Check mute state
muted=$(wpctl get-volume @DEFAULT_AUDIO_SINK@ | grep -c MUTED)
if [ "$muted" -gt 0 ]; then
  echo "volume:muted" > "$pipe"
else
  pct=$(wpctl get-volume @DEFAULT_AUDIO_SINK@ | awk '{print $2 * 100}')
  echo "volume:${pct%.*}" > "$pipe"
fi
```

```bash
chmod +x ~/qs-tutorial/myosd/scripts/osd-volume
```

## Step 4: Run it

Terminal 1:
```bash
qs -c ~/qs-tutorial/myosd
```

Terminal 2:
```bash
~/qs-tutorial/myosd/scripts/osd-volume up
~/qs-tutorial/myosd/scripts/osd-volume mute
```

---

## What you learned

- Named pipe IPC between scripts and QML
- Animated pill-shaped OSD window
- `WlrLayer.Overlay` for always-on-top display
- `Process` loop reading from a pipe
- State machine: type + value + muted → icon + bar + label
- Auto-hide timer
